pipeline {
  agent any

  stages {
    stage('Checkout') {
      steps {
        checkout scm
      }
    }

    stage('Install Dependencies') {
      steps {
        sh 'npm install'
      }
    }

    stage('Run Tests') {
      steps {
        sh 'npm test'
      }
    }

    stage('Build Docker Image') {
      steps {
        script {
          def imageName = "calculator-node"
          def imageTag = "latest"
          sh "docker build -t ${imageName}:${imageTag} ."
        }
      }
    }
  }
}
